    int x = getentityproperty(self, "x");
    int y = getentityproperty(self, "a");
    int Sx = getentityproperty(self, "xdir");
    int Sy = getentityproperty(self, "tossv");
    int XPos = openborvariant("xpos");
    int Screen = openborvariant("hResolution");

    if(x < XPos+30 && Sx < 0){
      changeentityproperty(self, "velocity", -Sx, 0, Sy);
    } else if(x > Screen+XPos-30 && Sx > 0){
      changeentityproperty(self, "velocity", -Sx, 0, Sy);
    } else if(y <= 5){
      changeentityproperty(self, "velocity", 0, 0, 1.5);
    } else if(y < 20 && Sy < 0){
      changeentityproperty(self, "velocity", Sx, 0, -Sy);
    } else if(y > 220 && Sy > 0){
      changeentityproperty(self, "velocity", Sx, 0, -Sy);
    }

    if(frame==1){
      int iR = rand()%20 + 20;
      int Vx; int Vy;

      if(iR >= 0 && iR < 5){ 
        Vx = 1.5;
        Vy = 0;
      } else if(iR >= 5 && iR < 10){ 
        Vx = 1.5;
        Vy = 1.5;
      } else if(iR >= 10 && iR < 15){
        Vx = 0;
        Vy = 1.5;
      } else if(iR >= 15 && iR < 20){
        Vx = -1.5;
        Vy = 1.5;
      } else if(iR >= 20 && iR < 25){
        Vx = -1.5;
        Vy = 0;
      } else if(iR >= 25 && iR < 30){
        Vx = -1.5;
        Vy = -1.5;
      } else if(iR >= 30 && iR < 35){
        Vx = 0;
        Vy = -1.5;
      } else { 
        Vx = 1.5;
        Vy = -1.5;
      }

      changeentityproperty(self, "velocity", Vx, 0, Vy);
    }
        return;
    }
    if(animhandle==705)
    {
        if(frame==1)

            suicide();
